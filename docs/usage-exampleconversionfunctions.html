Usage Example

 

// Example: CVTGF grain (G_float -> IEEE single)

// ASA: CVTGF converts VAX G to IEEE S; invalid on reserved operand; map to qNaN.

void CVTGF_InstructionGrain::execute(AlphaCPU* cpu, AlphaProcessorContext* ctx, quint64 rawInstr) const

{

    const quint64 pc = ctx->getPC();

    auto* status = ctx->exceptionStatusRegister();

    auto  fpcr   = ctx->fpcr();

 

    // Per ISA note: Fa must be F31 for some CVT forms; enforce if applicable to this variant.

    if (m_ra != 31) {

        ExceptionInfo ei; ei.setPcAtFault(pc);

        ei.category = ExceptionCategory::EMULATOR;

        ei.detail.emulator = EmulatorReason::UNHANDLED_ILLEGAL_INSTRUCTION;

        cpu->raiseException(ei);

        return;

    }

 

    const dt_gfloat srcG = ctx->readFpRegGFloat(m_rb);

    const auto mode = fpcr->getRoundingMode();

 

    const CvtfpConversionResult conv = checkCvtgfConversion(srcG, mode);

 

    if (conv.invalid)   status->setFlag(OptimizedExceptionStatusRegister::StatusBit::InvalidOpOccurred, true);

    if (conv.overflow)  status->setFlag(OptimizedExceptionStatusRegister::StatusBit::OverflowOccurred, true);

    if (conv.underflow) status->setFlag(OptimizedExceptionStatusRegister::StatusBit::UnderflowOccurred, true);

    if (conv.inexact)   status->setFlag(OptimizedExceptionStatusRegister::StatusBit::InexactOccurred, true);

 

    // If traps are enabled and any occurred, raise here instead of write-back.

    if (fpcr->trapOnAny(status)) {

        cpu->raiseFloatingPointException(pc, *status);

        return;

    }

 

    // Write back: conv.fpOut is a double containing the rounded single value.

    // The register bank API is assumed to accept a single-precision write.

    ctx->writeFpRegSingle(m_rc, static_cast<float>(conv.fpOut));

}
