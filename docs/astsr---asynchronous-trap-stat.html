ASTSR - Asynchronous Trap Status Register

ASTSR Helper Functions and Constants (Alpha AXP-EMulatR)

 

Considerations.

•**All helpers are SMP-safe, thread-safe, and match Alpha architectural AST model**

 

**Bit Constants & Masks**

------------------------

ASTSR_KERNEL_SHIFT         // Bit position for Kernel mode (0)

ASTSR_EXECUTIVE_SHIFT      // Bit position for Executive mode (1)

ASTSR_SUPERVISOR_SHIFT     // Bit position for Supervisor mode (2)

ASTSR_USER_SHIFT           // Bit position for User mode (3)

ASTSR_KERNEL_PENDING       // Mask: Kernel AST pending (0x01)

ASTSR_EXECUTIVE_PENDING    // Mask: Executive AST pending (0x02)

ASTSR_SUPERVISOR_PENDING   // Mask: Supervisor AST pending (0x04)

ASTSR_USER_PENDING         // Mask: User AST pending (0x08)

ASTSR_ALL_MODES_MASK       // Mask: All modes (0x0F)

 

ASTSR_NONE_PENDING         // No modes pending (0x00)

ASTSR_KERNEL_ONLY_PENDING  // Only Kernel pending (0x01)

ASTSR_EXEC_ONLY_PENDING    // Only Executive pending (0x02)

ASTSR_SUPER_ONLY_PENDING   // Only Supervisor pending (0x04)

ASTSR_USER_ONLY_PENDING    // Only User pending (0x08)

ASTSR_KERNEL_EXEC_PENDING  // Kernel & Executive pending (0x03)

ASTSR_KERNEL_SUPER_PENDING // Kernel & Supervisor pending (0x05)

ASTSR_KERNEL_USER_PENDING  // Kernel & User pending (0x09)

ASTSR_EXEC_SUPER_PENDING   // Executive & Supervisor pending (0x06)

ASTSR_EXEC_USER_PENDING    // Executive & User pending (0x0A)

ASTSR_SUPER_USER_PENDING   // Supervisor & User pending (0x0C)

ASTSR_KERNEL_EXEC_SUPER_PENDING // Kernel, Exec, Supervisor (0x07)

ASTSR_KERNEL_EXEC_USER_PENDING  // Kernel, Exec, User (0x0B)

ASTSR_KERNEL_SUPER_USER_PENDING // Kernel, Super, User (0x0D)

ASTSR_EXEC_SUPER_USER_PENDING   // Exec, Super, User (0x0E)

ASTSR_ALL_PENDING          // All modes pending (0x0F)

 

**Atomic Setters**

------------------------

setASTSRRaw(quint64 value)                   // Atomically set all pending bits

setASTSRMask(quint8 mask, bool pending)      // Set/clear bits (mask) atomically

setASTSRMaskDirect(quint8 mask)              // Atomically set bits to mask value

 

setASTSRNonePending()                        // Set no ASTs pending

setASTSRKernelOnlyPending()                  // Only Kernel pending

setASTSRExecOnlyPending()                    // Only Executive pending

setASTSRSuperOnlyPending()                   // Only Supervisor pending

setASTSRUserOnlyPending()                    // Only User pending

setASTSRKernelExecPending()                  // Kernel & Exec pending

setASTSRKernelSuperPending()                 // Kernel & Super pending

setASTSRKernelUserPending()                  // Kernel & User pending

setASTSRExecSuperPending()                   // Exec & Super pending

setASTSRExecUserPending()                    // Exec & User pending

setASTSRSuperUserPending()                   // Super & User pending

setASTSRKernelExecSuperPending()             // Kernel, Exec, Super pending

setASTSRKernelExecUserPending()              // Kernel, Exec, User pending

setASTSRKernelSuperUserPending()             // Kernel, Super, User pending

setASTSRExecSuperUserPending()               // Exec, Super, User pending

setASTSRAllPending()                         // All modes pending

clearAllASTPending()                         // Atomically clear all AST pending bits

 

setASTPendingKernel(bool pending)            // Atomically set/clear kernel pending

setASTPendingExecutive(bool pending)         // Atomically set/clear executive pending

setASTPendingSupervisor(bool pending)        // Atomically set/clear supervisor pending

setASTPendingUser(bool pending)              // Atomically set/clear user pending

 

**Atomic Getters**

------------------------

quint8 getASTSRMask() const                  // Read lower 4 bits (mask)

bool isASTPendingForMode(quint8 modeBit) const // Is AST pending for modeBit?

bool isASTPendingKernel() const              // Kernel pending?

bool isASTPendingExecutive() const           // Executive pending?

bool isASTPendingSupervisor() const          // Supervisor pending?

bool isASTPendingUser() const                // User pending?

bool isAnyASTPending(quint8 mask)            // Any AST pending (mask)?

 

bool isOnlyUserASTPending(quint8 mask)       // Only user pending?

bool isAllASTPending(quint8 mask)            // All modes pending?
