// ========== UPDATED AlphaFloatingPointExecutor INTEGRATION ==========

 

class AlphaFloatingPointExecutor : public QObject {

private:

   AlphaCPU* m_cpu;

   

   // ✅ BOTH TLB AND CACHE ARE NEEDED

   QSharedPointer<AlphaTranslationCache> m_iTLB;          // Address translation

   QSharedPointer<AlphaTranslationCache> m_dTLB;          // Address translation

   QSharedPointer<AlphaInstructionCache> m_instructionCache; // L1 I-Cache  

   QSharedPointer<UnifiedDataCache> m_level1DataCache;       // L1 D-Cache

   QSharedPointer<UnifiedDataCache> m_level2Cache;           // L2 Unified Cache

   QSharedPointer<UnifiedDataCache> m_level3Cache;           // L3 Shared Cache

 

public:

   // Attach BOTH TLB and Cache references

   void attachTranslationCache(QSharedPointer<AlphaTranslationCache> iTLB,

                              QSharedPointer<AlphaTranslationCache> dTLB) {

       m_iTLB = iTLB;

       m_dTLB = dTLB;

   }

   

   // Keep existing cache attachment methods

   void attachInstructionCache(QSharedPointer<AlphaInstructionCache> icache) {

       m_instructionCache = icache;

   }

   void attachLevel1DataCache(QSharedPointer<UnifiedDataCache> l1dcache) {

       m_level1DataCache = l1dcache;

   }

   void attachLevel2Cache(QSharedPointer<UnifiedDataCache> l2cache) {

       m_level2Cache = l2cache;

   }

   void attachLevel3Cache(QSharedPointer<UnifiedDataCache> l3cache) {

       m_level3Cache = l3cache;

   }

 

private:

   // ✅ TWO-STAGE MEMORY ACCESS: TLB THEN CACHE

   bool fetchInstructionWithCache(quint64 virtualPC, quint32& instruction) {

       QMutexLocker locker(&m_statsMutex);

       

       // STAGE 1: TLB Translation (Virtual → Physical)

       quint64 physicalPC;

       if (!m_iTLB->lookup(virtualPC, m_cpu->getCurrentASN(), false, true, physicalPC)) {

           m_l1ICacheMisses++;

           updateCacheStatistics("TLB-MISS", false);

           

           // Handle TLB miss (page fault simulation)

           handleInstructionTLBMiss(virtualPC);

           return false;

       }

       

       // STAGE 2: Cache Access (Physical address → Data)

       bool cacheHit = false;

       

       // Try L1 Instruction Cache first

       if (m_instructionCache) {

           cacheHit = m_instructionCache->read(physicalPC,

                                             reinterpret_cast<quint8*>(&instruction), 4);

           if (cacheHit) {

               m_l1ICacheHits++;

               updateCacheStatistics("L1I", true);

               return true;

           } else {

               m_l1ICacheMisses++;

               updateCacheStatistics("L1I", false);

           }

       }

       

       // Try L2 Cache

       if (m_level2Cache) {

           cacheHit = m_level2Cache->read(physicalPC,

                                        reinterpret_cast<quint8*>(&instruction), 4);

           if (cacheHit) {

               m_l2CacheHits++;

               updateCacheStatistics("L2", true);

               

               // Fill L1 cache

               if (m_instructionCache) {

                   m_instructionCache->write(physicalPC,

                                           reinterpret_cast<quint8*>(&instruction), 4);

               }

               return true;

           } else {

               m_l2CacheMisses++;

               updateCacheStatistics("L2", false);

           }

       }

       

       // Try L3 Cache

       if (m_level3Cache) {

           cacheHit = m_level3Cache->read(physicalPC,

                                        reinterpret_cast<quint8*>(&instruction), 4);

           if (cacheHit) {

               m_l3CacheHits++;

               updateCacheStatistics("L3", true);

               

               // Fill upper cache levels

               if (m_level2Cache) {

                   m_level2Cache->write(physicalPC,

                                      reinterpret_cast<quint8*>(&instruction), 4);

               }

               if (m_instructionCache) {

                   m_instructionCache->write(physicalPC,

                                           reinterpret_cast<quint8*>(&instruction), 4);

               }

               return true;

           } else {

               m_l3CacheMisses++;

               updateCacheStatistics("L3", false);

           }

       }

       

       // Final fallback: Main memory access

       return m_cpu->readMemory(physicalPC, reinterpret_cast<quint8*>(&instruction), 4);

   }

   

   // ✅ ASYNC VERSION WITH BOTH TLB AND CACHE

   QFuture<bool> asyncFetchInstruction(quint64 virtualPC) {

       return QtConcurrent::run([this, virtualPC]() -> bool {

           // STAGE 1: Async TLB lookup

           auto tlbFuture = m_iTLB->lookupAsync(virtualPC, m_cpu->getCurrentASN(),

                                              false, true);

           

           if (!tlbFuture.result()) {

               return false; // TLB miss

           }

           

           // STAGE 2: Physical address cache access

           quint64 physicalPC;

           if (m_iTLB->lookupLockFree(virtualPC, m_cpu->getCurrentASN(),

                                    false, true, physicalPC)) {

               

               quint32 instruction;

               return fetchFromCacheHierarchy(physicalPC, instruction);

           }

           

           return false;

       });

   }
