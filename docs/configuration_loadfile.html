// Configuration structure for SCSI devices

struct SCSIDeviceConfig {

   quint8 targetId;

   quint8 deviceType;

   QString vendor;

   QString product;

   QString revision;

   QString imagePath;

   QVariantMap deviceSpecific;

   bool enabled;

   

   // Device-specific configurations

   struct DiskConfig {

       quint32 blockSize = 512;

       quint64 sizeInMB = 100;

       bool writeProtected = false;

       QString geometry;  // "C/H/S" format

   };

   

   struct TapeConfig {

       QString tapeType = "DAT";       // DAT, DLT, LTO, etc.

       quint32 blockSize = 512;

       bool compression = true;

       quint32 capacity = 4000;        // MB

   };

   

   struct CDROMConfig {

       bool doorLocked = false;

       QString discType = "DATA";      // DATA, AUDIO, MIXED

       quint32 speed = 24;            // 24x speed

   };

};

bool SCSIConfigManager::loadConfiguration(const QString& configFile) {

   QFile file(configFile);

   if (!file.open(QIODevice::ReadOnly)) {

       qWarning() << "Cannot open SCSI config file:" << configFile;

       return false;

   }

   

   QByteArray data = file.readAll();

   QJsonDocument doc = QJsonDocument::fromJson(data);

   

   if (!doc.isObject()) {

       qWarning() << "Invalid SCSI configuration format";

       return false;

   }

   

   QJsonObject root = doc.object();

   QJsonObject scsiBus = root["scsi_bus"].toObject();

   QJsonArray devices = scsiBus["devices"].toArray();

   

   m_deviceConfigs.clear();

   

   for (const auto& deviceValue : devices) {

       QJsonObject deviceObj = deviceValue.toObject();

       SCSIDeviceConfig config = parseDeviceConfig(deviceObj);

       if (config.enabled) {

           m_deviceConfigs[config.targetId] = config;

       }

   }

   

   qDebug() << "Loaded" << m_deviceConfigs.size() << "SCSI device configurations";

   return true;

}

 

SCSIDevice* SCSIConfigManager::createDevice(const SCSIDeviceConfig& config) {

   SCSIDevice* device = nullptr;

   

   switch (config.deviceType) {

   case TYPE_DISK: {

       auto diskConfig = config.deviceSpecific;

       quint32 blockSize = diskConfig.value("block_size", 512).toUInt();

       quint64 sizeMB = diskConfig.value("size_mb", 100).toULongLong();

       

       device = SCSIDeviceFactory::createDiskDrive(config.imagePath, sizeMB, blockSize);

       break;

   }

   case TYPE_TAPE: {

       auto tapeConfig = config.deviceSpecific;

       QString tapeType = tapeConfig.value("tape_type", "DAT").toString();

       

       device = SCSIDeviceFactory::createTapeDrive(config.imagePath, tapeType);

       break;

   }

   case TYPE_CDROM: {

       device = SCSIDeviceFactory::createCDROMDrive(config.imagePath);

       break;

   }

   default:

       qWarning() << "Unknown SCSI device type:" << config.deviceType;

       return nullptr;

   }

   

   if (device) {

       qDebug() << "Created SCSI device:" << config.vendor << config.product

                << "at target" << config.targetId;

   }

   

   return device;

}

 

SCSIBus* SCSIConfigManager::createConfiguredBus() {

   SCSIBus* bus = new SCSIBus(this);

   

   for (auto it = m_deviceConfigs.begin(); it != m_deviceConfigs.end(); ++it) {

       quint8 targetId = it.key();

       const SCSIDeviceConfig& config = it.value();

       

       if (config.enabled) {

           SCSIDevice* device = createDevice(config);

           if (device) {

               bus->attachDevice(targetId, device);

               device->setParent(bus); // Transfer ownership

           }

       }

   }

   

   return bus;

}

       DATA_OUT,

       STATUS,

       MESSAGE_IN,

       MESSAGE_OUT

   };

   

   // Advanced bus operations

   // SCSI protocol features

   void enableSynchronousTransfer(bool enable) { m_synchronousEnabled = enable; }

   void setTransferRate(quint32 rateKBps) { m_transferRate = rateKBps; }

   

   // Bus arbitration
